// src/context/analyzer.ts
import type { FileContext, LineContext, RegionInfo, FileType } from './types.js';

export function buildFileContext(filePath: string, content: string): FileContext {
  const rawLines = content.split('\n');
  const language = detectLang(filePath);
  const fileType = detectFileType(filePath);
  const lines: LineContext[] = [];
  const tryCatchRegions: RegionInfo[] = [];
  const functionRegions: RegionInfo[] = [];
  const imports: string[] = [];

  let inBlockComment = false;
  let inDocstring = false;
  let inTemplateLiteral = false;
  let tryCatchStart = -1;
  let braceDepthAtTry = 0;
  let braceDepth = 0;

  for (let i = 0; i < rawLines.length; i++) {
    const line = rawLines[i];
    const trimmed = line.trim();

    // Track blank
    const isBlank = trimmed === '';

    // Track block comments
    let isComment = false;
    if (inBlockComment) {
      isComment = true;
      if (trimmed.includes('*/')) {
        inBlockComment = false;
      }
    } else if (inDocstring) {
      isComment = true;
      if (trimmed.includes('"""') || trimmed.includes("'''")) {
        inDocstring = false;
      }
    } else if (trimmed.startsWith('//') || trimmed.startsWith('#') || trimmed.startsWith('*')) {
      isComment = true;
    } else if (trimmed.startsWith('/*')) {
      isComment = true;
      if (!trimmed.includes('*/')) {
        inBlockComment = true;
      }
    } else if ((trimmed.startsWith('"""') || trimmed.startsWith("'''")) && language === 'python') {
      isComment = true;
      // Check if docstring closes on same line (e.g., """hello""")
      const quote = trimmed.slice(0, 3);
      const rest = trimmed.slice(3);
      if (!rest.includes(quote)) {
        inDocstring = true;
      }
    }

    // Track template literals (backtick strings spanning lines)
    let isString = false;
    if (inTemplateLiteral) {
      isString = true;
      if (trimmed.includes('`')) {
        inTemplateLiteral = false;
      }
    } else if (!isComment) {
      const backtickCount = (line.match(/(?<!\\)`/g) || []).length;
      if (backtickCount % 2 !== 0) {
        inTemplateLiteral = true;
      }
    }

    // Track imports
    const isImport = !isComment && (
      /^\s*import\s/.test(line) ||
      /^\s*from\s+\S+\s+import\s/.test(line) ||
      /\brequire\s*\(/.test(line)
    );

    if (isImport) {
      const fromMatch = line.match(/from\s+['"]([^'"]+)['"]/);
      const requireMatch = line.match(/require\s*\(\s*['"]([^'"]+)['"]\s*\)/);
      const importMatch = line.match(/import\s+['"]([^'"]+)['"]/);
      const mod = fromMatch?.[1] ?? requireMatch?.[1] ?? importMatch?.[1];
      if (mod && !imports.includes(mod)) {
        imports.push(mod);
      }
    }

    // Track brace depth for try-catch regions
    if (!isComment && !isString) {
      for (const ch of line) {
        if (ch === '{') braceDepth++;
        if (ch === '}') braceDepth--;
      }

      if (/\btry\s*\{/.test(trimmed) || trimmed === 'try {' || trimmed === 'try{') {
        tryCatchStart = i + 1;
        braceDepthAtTry = braceDepth;
      }

      if (tryCatchStart > 0 && braceDepth <= braceDepthAtTry - 1) {
        // Heuristic: look for catch on this line or next
        if (/\bcatch\b/.test(trimmed) || /\bcatch\b/.test(rawLines[i + 1]?.trim() ?? '')) {
          // We're past the try-catch block
        }
        tryCatchRegions.push({ start: tryCatchStart, end: i + 1 });
        tryCatchStart = -1;
      }
    }

    lines.push({ isComment, isString, isImport, isBlank });
  }

  // Close unclosed try-catch
  if (tryCatchStart > 0) {
    tryCatchRegions.push({ start: tryCatchStart, end: rawLines.length });
  }

  return { filePath, language, fileType, lines, tryCatchRegions, functionRegions, imports };
}

function detectFileType(filePath: string): FileType {
  if (filePath.includes('.test.') || filePath.includes('.spec.') || filePath.includes('__tests__')) return 'test';
  if (filePath.includes('.config.') || filePath.includes('config/') || filePath.endsWith('.env') || filePath.includes('.rc')) return 'config';
  if (filePath.includes('migration') || filePath.includes('migrate')) return 'migration';
  if (filePath.includes('.generated.') || filePath.includes('.gen.')) return 'generated';
  return 'source';
}

function detectLang(filePath: string): string {
  const ext = filePath.slice(filePath.lastIndexOf('.'));
  const map: Record<string, string> = {
    '.ts': 'typescript', '.tsx': 'typescript',
    '.js': 'javascript', '.jsx': 'javascript',
    '.py': 'python', '.rs': 'rust', '.go': 'go',
    '.java': 'java', '.rb': 'ruby', '.php': 'php',
    '.c': 'c', '.cpp': 'cpp', '.h': 'c', '.hpp': 'cpp',
    '.cs': 'csharp', '.swift': 'swift', '.kt': 'kotlin',
    '.sql': 'sql', '.sh': 'shell', '.bash': 'shell',
  };
  return map[ext] ?? 'unknown';
}

// src/rules/insecure-defaults.ts
import type { Rule, Finding } from './types.js';
import type { FileContext } from '../context/types.js';

// SEC006-SEC009: Patterns from TrailOfBits insecure-defaults skill
// Detects fail-open configurations that let apps run insecurely in production.

export const failOpenEnv: Rule = {
  id: 'SEC006',
  name: 'fail-open-env',
  category: 'security',
  severity: 'high',
  languages: ['typescript', 'javascript', 'python', 'ruby', 'go', 'java', 'php'],
  detect(lines, filePath, context?) {
    const findings: Finding[] = [];
    const secretKeywords = /secret|password|passwd|pwd|token|api[_-]?key|private[_-]?key|auth/i;

    for (let i = 0; i < lines.length; i++) {
      if (context?.lines[i]?.isComment || context?.lines[i]?.isString) continue;
      const line = lines[i];

      // JS/TS: process.env.X || 'fallback' or process.env.X ?? 'fallback'
      const jsEnvFallback = /process\.env\.(\w+)\s*(\|\||\\?\?)\s*['"](.+?)['"]/;
      // Python: os.environ.get('X', 'fallback') or os.getenv('X') or 'fallback'
      const pyEnvGet = /os\.environ\.get\s*\(\s*['"](\w+)['"]\s*,\s*['"](.+?)['"]\s*\)/;
      const pyGetenvOr = /os\.getenv\s*\(\s*['"](\w+)['"]\s*\)\s*or\s+['"](.+?)['"]/;

      let envName = '';
      let fallbackVal = '';

      const jsMatch = line.match(jsEnvFallback);
      const pyMatch = line.match(pyEnvGet);
      const pyOrMatch = line.match(pyGetenvOr);

      if (jsMatch) { envName = jsMatch[1]; fallbackVal = jsMatch[3]; }
      else if (pyMatch) { envName = pyMatch[1]; fallbackVal = pyMatch[2]; }
      else if (pyOrMatch) { envName = pyOrMatch[1]; fallbackVal = pyOrMatch[2]; }

      if (envName && fallbackVal && secretKeywords.test(envName)) {
        findings.push({
          rule: 'SEC006', severity: 'high', confidence: 85,
          category: 'security',
          message: `Fail-open default: ${envName} falls back to hardcoded value "${fallbackVal.slice(0, 20)}..." — app runs with insecure default if env var missing`,
          file: filePath, line: i + 1, snippet: lines[i],
          suggestion: 'Fail-secure: throw an error if security-critical env var is missing',
        });
      }
    }
    return findings;
  },
};
